function checkRareItems()
    local player = game.Players.LocalPlayer
    local rareItems = {"God's Chalice", "Fist of Darkness", "Library Key", "Water Key"}
    for _, itemName in pairs(rareItems) do
        if player.Backpack:FindFirstChild(itemName) or (player.Character and player.Character:FindFirstChild(itemName)) then
            return true
        end
    end
    return false
end

-- 5. Logic Farm R∆∞∆°ng
function doFarmChest()
    task.spawn(function()
        while _G.AutoFarmChest do
            if checkRareItems() then 
                _G.AutoFarmChest = false
                Rayfield:Notify({Title = "üéÅ ƒê√É C√ì ƒê·ªí QU√ù!", Content = "D·ª´ng farm ƒë·ªÉ b·∫£o v·ªá v·∫≠t ph·∫©m!", Duration = 10})
                break 
            end

            local char = game.Players.LocalPlayer.Character
            local root = char and char:FindFirstChild("HumanoidRootPart")
            if root then
                local targetChest = nil
                local minDist = math.huge
                
                for _, v in pairs(game.Workspace:GetChildren()) do
                    if v.Name:find("Chest") and v:IsA("BasePart") then
                        local dist = (v.Position - root.Position).Magnitude
                        if dist < minDist then
                            minDist = dist
                            targetChest = v
                        end
                    end
                end

                if targetChest then
                    _G.IsTeleporting = true
                    root.CFrame = targetChest.CFrame * CFrame.new(0, 2, 0)
                    task.wait(0.2)
                    _G.IsTeleporting = false
                end
            end
            task.wait(0.1)
        end
    end)
end
